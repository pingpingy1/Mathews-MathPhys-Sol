\begin{align*}
    W(x)
    &:= J_m(x) Y_m'(x) - J_m'(x) Y_m(x) \\
    &= \frac{
        J_m \left( \cos m\pi J_m'(x) - J_{-m}'(x) \right)
        - J_m'(x) \left( \cos m\pi J_m(x) - J_{-m}(x) \right)
    }{\sin m\pi} \\
    &= -\frac{J_m(x) J_{-m}'(x) - J_m'(x) J_{-m}(x)}{\sin m\pi} \\
    &= \frac{2}{\pi x}
\end{align*}
