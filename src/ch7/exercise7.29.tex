\begin{align*}
    \int_0^\infty dx e^{-sx} \confhypergeom(a; c; x)
    &= \frac{\Gamma(c)}{\Gamma(a)} \sum_{n = 0}^\infty
       \frac{\Gamma(a + n)}{\Gamma(c + n)}
       \inv{n!} \int_0^\infty dx x^n e^{-sx} \\
    &= \frac{\Gamma(c)}{\Gamma(a)} \sum_{n = 0}^\infty
       \frac{\Gamma(a + n)}{\Gamma(c + n)} s^{-(n + 1)} \\
    &= \inv{s} \confhypergeom \left( a; c; s^{-1} \right)
\end{align*}
