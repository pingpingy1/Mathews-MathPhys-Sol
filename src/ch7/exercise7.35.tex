\item
We compare the integrals obtained by \reffig~\ref{subfig:problem7-35:a1} and \ref{subfig:problem7-35:a2}.
\[
    x_1 = r + 2,\;
    x_2 = r - s = 2r - x_1
\]
\[
    r
    = \int_0^1 \frac{dy}{\sqrt{1 - y^2}}
    = \halfpi
\]
\[
    \therefore \sin (x) = \sin (\pi - x)
\]

\item
\[
    x_1 = r + s + t,\;
    x_2 = r - s + t = x_1 - 2s
\]
\[
    s
    = \int_{-1}^1 \frac{dy}{\sqrt{1 - y^2}}
    = \pi
\]
\[
    \therefore \sin(x) = \sin(x - 2\pi)
\]

\end{enumerate}
