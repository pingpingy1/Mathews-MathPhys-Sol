\begin{align*}
    \Lapltransf{f}
    &= \int_0^\infty dx f(x) e^{-sx} \\
    &= \sum_{n=0}^\infty \int_{2n}^{2n+1} dx e^{-sx} \\
    &= \inv{s} \sum_{n=0}^\infty \left( e^{-2ns} - e^{-(2n+1)s} \right) \\
    &= \inv{s} \left( \inv{1 - e^{-2s}} - \frac{e^{-s}}{1 - e^{-2s}} \right) \\
    &= \frac{e^{s/2}}{s} \frac{\sinh s/2}{\sinh s}
\end{align*}
